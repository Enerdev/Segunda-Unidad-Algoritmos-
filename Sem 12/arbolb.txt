package arbolb.indice;

import javax.swing.*;
import java.awt.*;
import java.util.*;
import javax.swing.table.DefaultTableModel;
import java.util.List;
import java.util.ArrayList;

public class SimuladorIndiceB extends JFrame {
    private BTree arbol;
    private JTable tabla;
    private DefaultTableModel tablaModelo;
    private JTextField entradaClave;
    private JTextArea salida;

    public SimuladorIndiceB(int grado) {
        super("Simulador de Índice con Árbol B (Orden " + grado + ")");
        setSize(900, 600);
        setDefaultCloseOperation(EXIT_ON_CLOSE);

        arbol = new BTree(grado);
        entradaClave = new JTextField(10);
        salida = new JTextArea(5, 40);
        salida.setEditable(false);

        JButton insertarBtn = new JButton("Insertar");
        JButton buscarBtn = new JButton("Buscar");
        JButton rangoBtn = new JButton("Buscar BETWEEN");

        JPanel controlPanel = new JPanel();
        controlPanel.add(new JLabel("ID:"));
        controlPanel.add(entradaClave);
        controlPanel.add(insertarBtn);
        controlPanel.add(buscarBtn);
        controlPanel.add(rangoBtn);

        tablaModelo = new DefaultTableModel(new String[]{"ID", "Nombre"}, 0);
        tabla = new JTable(tablaModelo);
        JScrollPane scrollTabla = new JScrollPane(tabla);

        JScrollPane scrollSalida = new JScrollPane(salida);

        JSplitPane splitPane = new JSplitPane(JSplitPane.VERTICAL_SPLIT, scrollTabla, scrollSalida);
        splitPane.setResizeWeight(0.5);

        add(controlPanel, BorderLayout.NORTH);
        add(splitPane, BorderLayout.CENTER);

        insertarBtn.addActionListener(e -> {
            try {
                int clave = Integer.parseInt(entradaClave.getText());
                String nombre = "Nombre" + clave;
                arbol.insert(clave);
                tablaModelo.addRow(new Object[]{clave, nombre});
                salida.setText("Insertado ID: " + clave);
            } catch (NumberFormatException ex) {
                salida.setText("ID inválido.");
            }
        });

        buscarBtn.addActionListener(e -> {
            try {
                int clave = Integer.parseInt(entradaClave.getText());
                boolean existe = arbol.search(clave);
                salida.setText(existe ? "ID encontrado." : "ID no encontrado.");
            } catch (Exception ex) {
                salida.setText("Error en búsqueda.");
            }
        });

        rangoBtn.addActionListener(e -> {
            String input = JOptionPane.showInputDialog(this, "Ingrese rango: inicio,fin");
            if (input != null && input.contains(",")) {
                try {
                    String[] partes = input.split(",");
                    int ini = Integer.parseInt(partes[0].trim());
                    int fin = Integer.parseInt(partes[1].trim());
                    List<Integer> encontrados = arbol.buscarRango(ini, fin);
                    salida.setText("Resultados BETWEEN " + ini + " y " + fin + ":\n" + encontrados);
                } catch (Exception ex) {
                    salida.setText("Error al interpretar el rango.");
                }
            }
        });
    }

    public static void main(String[] args) {
        String gradoStr = JOptionPane.showInputDialog(null, "Ingrese el orden del Árbol B (ej: 3, 4, 5):");
        try {
            int g = Integer.parseInt(gradoStr);
if (g < 3) g = 3;
final int grado = g;
            SwingUtilities.invokeLater(new Runnable() {
    public void run() {
        new SimuladorIndiceB(grado).setVisible(true);
    }
});
        } catch (Exception e) {
            JOptionPane.showMessageDialog(null, "Entrada inválida. Cerrando aplicación.");
        }
    }
}

// --------- Árbol B básico ---------

class BTreeNode {
    int t;
    ArrayList<Integer> claves;
    ArrayList<BTreeNode> hijos;
    boolean hoja;

    BTreeNode(int t, boolean hoja) {
        this.t = t;
        this.hoja = hoja;
        this.claves = new ArrayList<>();
        this.hijos = new ArrayList<>();
    }

    void insertarNoLleno(int clave) {
        int i = claves.size() - 1;

        if (hoja) {
            claves.add(0);
            while (i >= 0 && clave < claves.get(i)) {
                claves.set(i + 1, claves.get(i));
                i--;
            }
            claves.set(i + 1, clave);
        } else {
            while (i >= 0 && clave < claves.get(i)) i--;
            i++;
            if (hijos.get(i).claves.size() == 2 * t - 1) {
                dividirHijo(i);
                if (clave > claves.get(i)) i++;
            }
            hijos.get(i).insertarNoLleno(clave);
        }
    }

    void dividirHijo(int i) {
        BTreeNode y = hijos.get(i);
        BTreeNode z = new BTreeNode(y.t, y.hoja);

        for (int j = 0; j < t - 1; j++) {
            if (y.claves.size() > t) {
                z.claves.add(y.claves.remove(t));
            }
        }
        if (!y.hoja) {
            for (int j = 0; j < t; j++) {
                if (y.hijos.size() > t) {
                    z.hijos.add(y.hijos.remove(t));
                }
            }
        }

        hijos.add(i + 1, z);
        if (y.claves.size() > t - 1) {
            claves.add(i, y.claves.remove(t - 1));
        }
    }

    boolean buscar(int clave) {
        int i = 0;
        while (i < claves.size() && clave > claves.get(i)) i++;
        if (i < claves.size() && claves.get(i) == clave) return true;
        if (hoja) return false;
        return hijos.get(i).buscar(clave);
    }

    void buscarRango(int ini, int fin, List<Integer> resultado) {
        int i = 0;
        while (i < claves.size() && claves.get(i) < ini) i++;

        for (; i < claves.size() && claves.get(i) <= fin; i++) {
            if (!hoja) hijos.get(i).buscarRango(ini, fin, resultado);
            resultado.add(claves.get(i));
        }
        if (!hoja && i < hijos.size()) hijos.get(i).buscarRango(ini, fin, resultado);
    }
}

class BTree {
    BTreeNode raiz;
    int t;

    BTree(int t) {
        this.t = t;
        raiz = new BTreeNode(t, true);
    }

    void insert(int clave) {
        if (raiz.claves.size() == 2 * t - 1) {
            BTreeNode nueva = new BTreeNode(t, false);
            nueva.hijos.add(raiz);
            nueva.dividirHijo(0);
            int i = (clave < nueva.claves.get(0)) ? 0 : 1;
            nueva.hijos.get(i).insertarNoLleno(clave);
            raiz = nueva;
        } else {
            raiz.insertarNoLleno(clave);
        }
    }

    boolean search(int clave) {
        return raiz.buscar(clave);
    }

    List<Integer> buscarRango(int ini, int fin) {
        List<Integer> resultado = new ArrayList<>();
        raiz.buscarRango(ini, fin, resultado);
        return resultado;
    }
}
